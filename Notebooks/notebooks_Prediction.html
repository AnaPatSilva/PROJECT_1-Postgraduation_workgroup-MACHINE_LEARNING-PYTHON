from sklearn import metrics
# Plot confusion matrix
def plot_confusion_matrix(fit_function, actual, predicted):
    cm = confusion_matrix(actual, predicted, labels=fit_function.classes_)
    # sns.heatmap(cm, annot=True, fmt='g', xticklabels=['True', 'False'], yticklabels=['True', 'False'])
    # cm_display = metrics.ConfusionMatrixDisplay(confusion_matrix=cm, display_labels=[False, True])
    cm_display = metrics.ConfusionMatrixDisplay(confusion_matrix=cm, display_labels= fit_function.classes_ )
    cm_display.plot()
    plt.ylabel('True label', fontsize=9)
    plt.xlabel('Predicited label', fontsize=9)
    # plt.title('Confusion matrix', fontsize=12).format()
    return

def model_metrics(actual, predicted):
    # Finding precision and recall
    accuracy = accuracy_score(actual, predicted)
    print("Accuracy   :", round(accuracy*100, 2))
    precision = precision_score(actual, predicted)
    print("Precision :", round(precision*100, 2))
    recall = recall_score(actual, predicted, average="binary")
    print("Recall    :", round(recall*100,2))
    F1_score = f1_score(actual, predicted, average="binary")
    print("F1-score  :", round(F1_score*100, 2))
    return

def conf_matrix(y_test, pred_test):
    # Creating a confusion matrix
    con_mat = confusion_matrix(y_test, pred_test)
    con_mat = pd.DataFrame(con_mat, range(2), range(2))

    sns.heatmap(con_mat, annot=True, annot_kws={"size": 16}, fmt='g', cmap='Blues', cbar=False)
